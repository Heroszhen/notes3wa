C’est une “façon” de gérer les traitements asynchrones avec une API unique en utilisant les callbacks.
let p = new Promise(function (fulfill, reject) {
   setTimeout(() => {
      fulfill('ACCOMPLIE !');
   }, 300);
});

p.then((value) => {
   console.log(value);
});
